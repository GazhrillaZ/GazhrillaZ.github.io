canvas

Canvas 对象表示一个 HTML 画布元素 - "canvas>"。它没有自己的行为，但是定义了一个 API 支持脚本化客户端绘图操作。

			var canv = document.getElementById('canv');
			var ctx = canv.getContext('2d');
			
			ctx.fillStyle="blue";
			ctx.fillRect(10,10,100,100);
			
			ctx.rect(10,10,100,100);
			ctx.stroke();
			
			ctx.fillStyle="red";
			ctx.beginPath();
			ctx.fillStyle="greenyellow";
			ctx.arc(100,100,50,0,Math.PI);
			ctx.closePath();
			ctx.fillStyle="aqua";
			ctx.fill();
			ctx.fillStyle="gold";
			ctx.stroke();
			
			ctx.beginPath();
			ctx.arc(76,111,3,0,2*Math.PI);
			ctx.fillStyle="black";
			ctx.fill();
			ctx.stroke();
			
			ctx.lineWidth=10;
			ctx.beginPath();
			ctx.moveTo(10,10);
			ctx.bezierCurveTo(10,200,210,200,210,10);
			ctx.lineTo(210,200);
			ctx.lineTo(10,200);
			ctx.lineTo(10,10);
			ctx.lineTo(215,10);
			ctx.closePath();
			ctx.stroke();
			
			var grd = ctx.createLinearGradient(0,0,175,50);
			grd.addColorStop(0,"#fff");
			grd.addColorStop(1,"#000");
			ctx.fillStyle=grd;
			ctx.fillRect(0,0,175,50);
