演示一

 
			$(document).ready(function(){
				$(".tab li").mouseenter(function(){
					$(this).addClass("active").siblings().removeClass("active")
					var idx=$(this).index()
					$(".main").eq(idx).addClass("selected").siblings().removeClass("selected")
				})
			})

演示二

 
			$(document).ready(function(){
		var zidx= 0;
		$("ol li").mouseenter(function(){
			zidx+=1;
			$(this).addClass("active").siblings().removeClass("active");
			var idx = $(this).index();
			$("ul li").eq(idx).css({
				"left":600,
				"z-index":zidx
			}).animate({
				"left":0
			},300);
		})
	})

演示三

 
			$(function () {
            $(".wrap li").each(function (i,ele) {
                $(this).css("background-image","url(img/" + (i + 1) +".jpg)");
            }).mouseenter(function () {
                $(this).stop().animate({
                    "width":600
                },300).siblings().stop().animate({
                    "width":200
                },300)
            }).mouseleave(function () {
                $(this).stop().animate({
                    "width":100
                },300)
            });
        })

演示七

 
			$(window).scroll(function() {
				var scrtop = $(document).scrollTop();
				console.log(scrtop)
				if(scrtop > 100 && scrtop < 2000) {
					$("#bar").fadeIn().offset({
						top: scrtop + 50,
						left: 300
					})
				} else {
					$("#bar").fadeOut()
				}

			});
