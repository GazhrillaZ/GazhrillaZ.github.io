string对象

调用

			var n="1222223";
			n.indexOf(searchvalue,fromindex)//字符,第几个
			n.match(searchvalue/regexp)//match() 方法将检索字符串 stringObject，以找到一个或多个与 regexp 匹配的文本
			n.slice(start,end)//开始,结束
			n.split(separator,howmany)//正则,返回值最大长度
			n.substr(start,length)//开始,长度

Array的调用

			var arr=[];
			arr.push("1");
			console.log(arr.push("2"));//返回长度
			console.log(arr);
			console.log(arr.unshift("0"));//返回长度
			console.log(arr);
			console.log(arr.shift());//返回元素
			console.log(arr);
			console.log(arr.pop());//返回元素
			console.log(arr);
			
			var arr = [1,2,3,4,5];
			var str = "12345"
			console.log(arr.slice(1,3));//2,3
			console.log(str.slice(1,3));//2,3
			console.log(arr);//1,2,3,4,5
